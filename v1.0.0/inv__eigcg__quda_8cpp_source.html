145  Block<MatrixXcd>(args.ritzVecs.derived(), 0, k, m-1, k) = es_tm1.eigenvectors().leftCols(k);

319

321  ComputeRitz<libtype::magma_lib>(args);
322  } else if( param.extlib_type == QUDA_EIGEN_EXTLIB ) {

329
