123  if (idxType == EVEN_X ) { // X even
124  // t = idx / XYZ;
125  // z = (idx / XY) % Z;
126  // y = (idx / X) % Y;
127  // idx += (parity + t + z + y) & 1;
128  // x = idx % X;

155  y = (idx / LX) % LY;
