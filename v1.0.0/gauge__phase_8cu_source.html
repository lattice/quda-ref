62  } else if (dim == 1) {
63  phase = (1.0 - 2.0 * ((t + x) % 2) );
64  } else if (dim == 2) {
65  phase = (1.0 - 2.0 * ((t + x + y) % 2) );
66  } else if (dim == 3) { // also apply boundary condition

68  }
69  } else if (phaseType == QUDA_STAGGERED_PHASE_TIFR) {

74  } else if (dim == 2) {

76  } else if (dim == 3) { // also apply boundary condition

80  if (dim==0) {
