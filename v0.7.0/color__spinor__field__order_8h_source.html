239 
240  // the length of the block on the last grid site might not extend to all threads
241  const int block_dim = (blockIdx.x == gridDim.x-1) ?
242  volume - (gridDim.x-1)*blockDim.x : blockDim.x;
243 
244  extern __shared__ typename mapper<Float>::type s_data[];
245 
246  int x0 = x-tid; // x0 is the base index from where we are reading
247  int i=tid;
248  while (i<vec_length*block_dim) {
249  int space_idx = i / vec_length;
250  int internal_idx = i - space_idx*vec_length;
251  int sh_idx = internal_idx*(blockDim.x+1) + space_idx;
252  s_data[sh_idx] = field[x0*vec_length + i];
