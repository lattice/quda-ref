00564       errorQuda("Staggered code only supports QUDA_MASS_NORMALIZATION");
00565     }
00566     return;
00567   }
00568 
00569   // multiply the source to compensate for normalization of the Dirac operator, if necessary
00570   switch (solution_type) {
00571   case QUDA_MAT_SOLUTION:
00572     if (mass_normalization == QUDA_MASS_NORMALIZATION ||
00573         mass_normalization == QUDA_ASYMMETRIC_MASS_NORMALIZATION) {
00574       coeff *= 2.0*kappa;
00575     }
00576     break;
00577   case QUDA_MATDAG_MAT_SOLUTION:
00578     if (mass_normalization == QUDA_MASS_NORMALIZATION ||
00579         mass_normalization == QUDA_ASYMMETRIC_MASS_NORMALIZATION) {
00580       coeff *= 4.0*kappa*kappa;
00581     }
00582     break;
00583   case QUDA_MATPC_SOLUTION:
00584     if (mass_normalization == QUDA_MASS_NORMALIZATION) {
00585       coeff *= 4.0*kappa*kappa;
00586     } else if (mass_normalization == QUDA_ASYMMETRIC_MASS_NORMALIZATION) {
00587       coeff *= 2.0*kappa;
00588     }
00589     break;
00590   case QUDA_MATPCDAG_MATPC_SOLUTION:
00591     if (mass_normalization == QUDA_MASS_NORMALIZATION) {
00592         coeff*=16.0*pow(kappa,4);
00593     } else if (mass_normalization == QUDA_ASYMMETRIC_MASS_NORMALIZATION) {
00594         coeff*=4.0*kappa*kappa;
00595     }
00596     break;
00597   default:
00598     errorQuda("Solution type %d not supported", solution_type);
00599   }
00600 
00601   if (verbosity >= QUDA_DEBUG_VERBOSE) printfQuda("Mass rescale done\n");   
00602 }

00756 void invertQuda(void *hp_x, void *hp_b, QudaInvertParam *param)
00757 {
00758   checkInvertParam(param);
00759   verbosity = param->verbosity;
00760 
00761   bool pc_solve = (param->solve_type == QUDA_DIRECT_PC_SOLVE ||
00762                    param->solve_type == QUDA_NORMEQ_PC_SOLVE);
00763 
00764   bool pc_solution = (param->solution_type == QUDA_MATPC_SOLUTION ||
00765                       param->solution_type == QUDA_MATPCDAG_MATPC_SOLUTION);
00766 
00767   param->spinorGiB = cudaGaugePrecise.volume * spinorSiteSize;
00768   if (!pc_solve) param->spinorGiB *= 2;
00769   param->spinorGiB *= (param->cuda_prec == QUDA_DOUBLE_PRECISION ? sizeof(double) : sizeof(float));
00770   if (param->preserve_source == QUDA_PRESERVE_SOURCE_NO) {
00771     param->spinorGiB *= (param->inv_type == QUDA_CG_INVERTER ? 5 : 7)/(double)(1<<30);
00772   } else {
00773     param->spinorGiB *= (param->inv_type == QUDA_CG_INVERTER ? 8 : 9)/(double)(1<<30);
00774   }
00775 
00776   param->secs = 0;
00777   param->gflops = 0;
00778   param->iter = 0;
00779 
00780   // create the dirac operator
00781   DiracParam diracParam;
00782   createDirac(diracParam, *param, pc_solve);
00783   Dirac &dirac = *d;
00784   Dirac &diracSloppy = *dSloppy;
00785 
00786   cpuColorSpinorField *h_b = NULL;
00787   cpuColorSpinorField *h_x = NULL;
00788   cudaColorSpinorField *b = NULL;

00790   cudaColorSpinorField *in = NULL;

00792 
00793   int *X = param->dslash_type == QUDA_ASQTAD_DSLASH ? 
00794     cudaFatLinkPrecise.X : cudaGaugePrecise.X;
00795 
00796   // wrap CPU host side pointers
00797   ColorSpinorParam cpuParam(hp_b, *param, X, pc_solution);
00798   h_b = new cpuColorSpinorField(cpuParam);
00799   cpuParam.v = hp_x;
00800   h_x = new cpuColorSpinorField(cpuParam);
00801     
00802   // download source
00803   ColorSpinorParam cudaParam(cpuParam, *param);     
00804   cudaParam.create = QUDA_COPY_FIELD_CREATE;
00805   b = new cudaColorSpinorField(*h_b, cudaParam); 
00806 
00807   if (param->use_init_guess == QUDA_USE_INIT_GUESS_YES) { // download initial guess
00808     x = new cudaColorSpinorField(*h_x, cudaParam); // solution  
00809   } else { // zero initial guess

00822   dirac.prepare(in, out, *x, *b, param->solution_type);
00823   if (param->verbosity >= QUDA_VERBOSE) {
00824     double nin = norm2(*in);
00825     printfQuda("Prepared source = %f\n", nin);   
00826   }
00827 
00828   massRescale(param->dslash_type, diracParam.kappa, param->solution_type, param->mass_normalization, *in);
00829

00831   case QUDA_CG_INVERTER:
00832     if (param->solution_type != QUDA_MATDAG_MAT_SOLUTION && param->solution_type != QUDA_MATPCDAG_MATPC_SOLUTION) {
00833       copyCuda(*out, *in);
00834       dirac.Mdag(*in, *out);
00835     }
00836     invertCgCuda(DiracMdagM(dirac), DiracMdagM(diracSloppy), *out, *in, param);
00837     break;
00838   case QUDA_BICGSTAB_INVERTER:
00839     if (param->solution_type == QUDA_MATDAG_MAT_SOLUTION || param->solution_type == QUDA_MATPCDAG_MATPC_SOLUTION) {
00840       invertBiCGstabCuda(DiracMdag(dirac), DiracMdag(diracSloppy), *out, *in, param);
00841       copyCuda(*in, *out);
00842     }
00843     invertBiCGstabCuda(DiracM(dirac), DiracM(diracSloppy), *out, *in, param);
00844     break;
00845   default:
00846     errorQuda("Inverter type %d not implemented", param->inv_type);
00847   }
00848

00850    double nx = norm2(*x);
00851    printfQuda("Solution = %f\n",nx);

00854   
00855   x->saveCPUSpinorField(*h_x); // since this is a reference, this won't work: h_x = x;
00856   
00857   if (param->verbosity >= QUDA_VERBOSE){
00858     double nx = norm2(*x);

00867     diracTune = false;
00868   }  
00869 
00870   delete h_b;
00871   delete h_x;
00872   delete b;
00873   delete x;
00874   
00875   return;

00884 void invertMultiShiftQuda(void **_hp_x, void *_hp_b, QudaInvertParam *param,
00885                           double* offsets, int num_offsets, double* residue_sq)
00886 {
00887   checkInvertParam(param);
00888   verbosity = param->verbosity;
00889 
00890   // Are we doing a preconditioned solve */
00891   /* What does NormEq solve mean in the shifted case? 
00892    */
00893   if (param->solve_type != QUDA_NORMEQ_PC_SOLVE &&
00894       param->solve_type != QUDA_NORMEQ_SOLVE) { 
00895     errorQuda("Direct solve_type is not supported in invertMultiShiftQuda()\n");
00896   }
00897 
00898   bool pc_solve = (param->solve_type == QUDA_NORMEQ_PC_SOLVE);
00899 
00900   // In principle one can do a MATPC Solution for a hermitian M_pc
00901   // In practice most of the time I guess one will do a M^\dagger_pc M_pc solution.
00902   bool pc_solution = (param->solution_type == QUDA_MATPC_SOLUTION ||
00903                       param->solution_type == QUDA_MATPCDAG_MATPC_SOLUTION );
00904 
00905   // No of GiB in a checkerboard of a spinor
00906   param->spinorGiB = cudaGaugePrecise.volume * spinorSiteSize;
00907   if( !pc_solve) param->spinorGiB *= 2; // Double volume for non PC solve
00908   
00909   // **** WARNING *** this may not match implementation... 
00910   if( param->inv_type == QUDA_CG_INVERTER ) { 
00911     // CG-M needs 5 vectors for the smallest shift + 2 for each additional shift
00912     param->spinorGiB *= (5 + 2*(num_offsets-1))/(double)(1<<30);
00913   }
00914   else {
00915     // BiCGStab-M needs 7 for the original shift + 2 for each additional shift + 1 auxiliary
00916     // (Jegerlehner hep-lat/9612014 eq (3.13)
00917     param->spinorGiB *= (7 + 2*(num_offsets-1))/(double)(1<<30);
00918   }
00919 
00920   // Timing and FLOP counters
00921   param->secs = 0;
00922   param->gflops = 0;
00923   param->iter = 0;
00924   
00925   // Find the smallest shift and its offset.
00926   double low_offset = offsets[0];
00927   int low_index = 0;
00928   for (int i=1;i < num_offsets;i++){
00929     if (offsets[i] < low_offset){
00930       low_offset = offsets[i];
00931       low_index = i;
00932     }
00933   }
00934   
00935   // Host pointers for x, take a copy of the input host pointers
00936   void** hp_x;
00937   hp_x = new void* [ num_offsets ];
00938 
00939   void* hp_b = _hp_b;
00940   for(int i=0;i < num_offsets;i++){
00941     hp_x[i] = _hp_x[i];
00942   }
00943   
00944   // Now shift things so that the vector with the smallest shift 
00945   // is in the first position of the array
00946   if (low_index != 0){
00947     void* tmp = hp_x[0];
00948     hp_x[0] = hp_x[low_index] ;
00949     hp_x[low_index] = tmp;
00950     
00951     double tmp1 = offsets[0];
00952     offsets[0]= offsets[low_index];
00953     offsets[low_index] =tmp1;
00954   }
00955     
00956   // Create the matrix.
00957   // The way this works is that createDirac will create 'd' and 'dSloppy'
00958   // which are global. We then grab these with references...
00959   //
00960   // Balint: Isn't there a  nice construction pattern we could use here? This is 
00961   // expedient but yucky.
00962   DiracParam diracParam; 
00963   if (param->dslash_type == QUDA_ASQTAD_DSLASH){
00964     param->mass = sqrt(offsets[0]/4);  
00965   }
00966   createDirac(diracParam, *param, pc_solve);
00967   Dirac &dirac = *d;
00968   Dirac &diracSloppy = *dSloppy;
00969 
00970   cpuColorSpinorField *h_b = NULL; // Host RHS
00971   cpuColorSpinorField **h_x = NULL;
00972   cudaColorSpinorField *b = NULL;   // Cuda RHS
00973   cudaColorSpinorField **x = NULL;  // Cuda Solutions
00974 
00975   // Grab the dimension array of the input gauge field.
00976   int *X = ( param->dslash_type == QUDA_ASQTAD_DSLASH ) ? 
00977     cudaFatLinkPrecise.X : cudaGaugePrecise.X;
00978 
00979   // Wrap CPU host side pointers
00980   // 
00981   // Balint: This creates a ColorSpinorParam struct, from the host data pointer, 
00982   // the definitions in param, the dimensions X, and whether the solution is on 
00983   // a checkerboard instruction or not. These can then be used as 'instructions' 
00984   // to create the actual colorSpinorField
00985   ColorSpinorParam cpuParam(hp_b, *param, X, pc_solution);
00986   h_b = new cpuColorSpinorField(cpuParam);
00987 
00988   h_x = new cpuColorSpinorField* [ num_offsets ]; // DYNAMIC ALLOCATION
00989   for(int i=0; i < num_offsets; i++) { 
00990     cpuParam.v = hp_x[i];
00991     h_x[i] = new cpuColorSpinorField(cpuParam);
00992   }
00993 
00994   // Now I need a colorSpinorParam for the device
00995   ColorSpinorParam cudaParam(cpuParam, *param);
00996   // This setting will download a host vector
00997   cudaParam.create = QUDA_COPY_FIELD_CREATE;
00998   b = new cudaColorSpinorField(*h_b, cudaParam); // Creates b and downloads h_b to it
00999 
01000   // Create the solution fields filled with zero
01001   x = new cudaColorSpinorField* [ num_offsets ];
01002   cudaParam.create = QUDA_ZERO_FIELD_CREATE;
01003   for(int i=0; i < num_offsets; i++) { 
01004     x[i] = new cudaColorSpinorField(cudaParam);
01005   }
01006 
01007   // Check source norms
01008   if( param->verbosity >= QUDA_VERBOSE ) {
01009     double nh_b = norm2(*h_b);
01010     double nb = norm2(*b);
01011     printfQuda("Source: CPU= %f, CUDA copy = %f\n", nh_b,nb);
01012   }
01013 
01014   // tune the Dirac Kernel
01015   tuneDirac(*param, pc_solution ? *(x[0]) : (x[0])->Even());
01016   
01017   
01018   massRescale(param->dslash_type, diracParam.kappa, param->solution_type, param->mass_normalization, *b);
01019   double *rescaled_shifts = new double [num_offsets];
01020   for(int i=0; i < num_offsets; i++){ 
01021     rescaled_shifts[i] = offsets[i];
01022     massRescaleCoeff(param->dslash_type, diracParam.kappa, param->solution_type, param->mass_normalization, rescaled_shifts[i]);
01023   }
01024   invertMultiShiftCgCuda(DiracMdagM(dirac), DiracMdagM(diracSloppy), x, *b, param, rescaled_shifts, num_offsets, residue_sq);
01025   
01026   
01027   delete [] rescaled_shifts;
01028 
01029   for(int i=0; i < num_offsets; i++) { 
01030     x[i]->saveCPUSpinorField(*h_x[i]);
01031   }
01032 
01033   for(int i=0; i < num_offsets; i++){ 
01034     delete h_x[i];
01035     delete x[i];
01036   }
01037 
01038   delete h_b;
01039   delete b;
01040 
01041   delete [] h_x;
01042   delete [] x;
01043 
01044   delete [] hp_x;
01045 
01046   if (!param->preserve_dirac) {
01047     delete d;
01048     delete dSloppy;
01049     diracCreation = false;
01050     diracTune = false;
01051   }  
01052 
01053   return;
01054 }
