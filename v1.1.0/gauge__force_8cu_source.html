152  linkA = linkA * linkB;
153  dx[lnkdir]++; // now have to update to new location
154  nbr_oddbit = nbr_oddbit^1;
155  } else {

157  nbr_oddbit = nbr_oddbit^1;
158  linkB = arg.u(lnkdir, linkIndexShift(x,dx,arg.E), nbr_oddbit);

161  } //j
