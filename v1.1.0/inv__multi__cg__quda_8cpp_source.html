132  alpha[0] = r2[0] / pAp;
133  zeta[0] = 1.0;

135  double c0 = zeta[j] * zeta_old[j] * alpha_old[j_low];
136  double c1 = alpha[j_low] * beta[j_low] * (zeta_old[j]-zeta[j]);

142  }
143  else {
144  zeta[j] = 0.0;
145  }
146  if (zeta[j] != 0.0){
147  alpha[j] = alpha[j_low] * zeta[j] / zeta_old[j];

149  else {
