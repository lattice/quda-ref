242  Matrix<T,2> a;
243  T aabs, ctheta, stheta, phi;
244  a(0,0) = Random<T>(localState, (T)-1.0, (T)1.0);
245  aabs = sqrt( 1.0 - a(0,0) * a(0,0));
246  ctheta = Random<T>(localState, (T)-1.0, (T)1.0);
