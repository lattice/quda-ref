692  cudaStream_t *stream) {
693 
694 #ifdef MULTI_GPU
695  int Nvec = (nSpin == 1 || precision == QUDA_DOUBLE_PRECISION) ? 2 : 4;
696  int Nint = (nColor * nSpin * 2) / (nSpin == 4 ? 2 : 1); // (spin proj.) degrees of freedom
697  int Npad = Nint / Nvec; // number Nvec buffers we have

700

702

712  const int Nt_minus1_offset = (volumeCB - nFace*ghostFace[3])/x4; // N_t -1 = Vh-Vsh
713 
714  int offset = 0;
715  if (nSpin == 1) {
716  offset = (dir == QUDA_BACKWARDS) ? 0 : Nt_minus1_offset;
717  } else if (nSpin == 4) {
718  // !dagger: send lower components backwards, send upper components forwards
719  // dagger: send upper components backwards, send lower components forwards
720  bool upper = dagger ? true : false; // Fwd is !Back
721  if (dir == QUDA_FORWARDS) upper = !upper;
722  int lower_spin_offset = Npad*stride;
723  if (upper) offset = (dir == QUDA_BACKWARDS ? 0 : Nt_minus1_offset);
724  else offset = lower_spin_offset + (dir == QUDA_BACKWARDS ? 0 : Nt_minus1_offset);
725  }
726 
727  size_t len = nFace*(ghostFace[3]/x4)*Nvec*precision;
728  size_t dpitch = x4*len;
729  size_t spitch = stride*Nvec*precision;

731  // QUDA Memcpy NPad's worth.
732  // -- Dest will point to the right beginning PAD.
733  // -- Each Pad has size Nvec*Vsh Floats.
734  // -- There is Nvec*Stride Floats from the start of one PAD to the start of the next

736  void *dst = (char*)ghost_spinor + s*len;
737  void *src = (char*)v + (offset + s*(volumeCB/x4))*Nvec*precision;
738  qudaMemcpy2DAsync(dst, dpitch, src, spitch, len, Npad, cudaMemcpyDeviceToHost, *stream);
739

741  size_t len = nFace*(ghostFace[3]/x4)*sizeof(float);
742  int norm_offset = (dir == QUDA_BACKWARDS) ? 0 : Nt_minus1_offset*sizeof(float);
743  void *dst = (char*)ghost_spinor + nFace*Nint*ghostFace[3]*precision + s*len;
744  void *src = (char*)norm + norm_offset + s*(volumeCB/x4)*sizeof(float);
745  qudaMemcpyAsync(dst, src, len, cudaMemcpyDeviceToHost, *stream);
746  }
747  }
748  }else{
749  int flavorVolume = volume / 2;
750  int flavorTFace = ghostFace[3] / 2;
751  int flavor1_Nt_minus1_offset = (flavorVolume - flavorTFace);
752  int flavor2_Nt_minus1_offset = (volume - flavorTFace);
753  int flavor1_offset = 0;
754  int flavor2_offset = 0;
755  // !dagger: send lower components backwards, send upper components forwards
756  // dagger: send upper components backwards, send lower components forwards
757  bool upper = dagger ? true : false; // Fwd is !Back
758  if (dir == QUDA_FORWARDS) upper = !upper;
759  int lower_spin_offset = Npad*stride;//ndeg tm: stride=2*flavor_volume+pad
760  if (upper) {
761  flavor1_offset = (dir == QUDA_BACKWARDS ? 0 : flavor1_Nt_minus1_offset);
762  flavor2_offset = (dir == QUDA_BACKWARDS ? flavorVolume : flavor2_Nt_minus1_offset);
763  }else{
764  flavor1_offset = lower_spin_offset + (dir == QUDA_BACKWARDS ? 0 : flavor1_Nt_minus1_offset);
765  flavor2_offset = lower_spin_offset + (dir == QUDA_BACKWARDS ? flavorVolume : flavor2_Nt_minus1_offset);
766  }
767 
768  // QUDA Memcpy NPad's worth.
769  // -- Dest will point to the right beginning PAD.
770  // -- Each Pad has size Nvec*Vsh Floats.
771  // -- There is Nvec*Stride Floats from the start of one PAD to the start of the next
772 
773  void *dst = (char*)ghost_spinor;
774  void *src = (char*)v + flavor1_offset*Nvec*precision;
775  size_t len = flavorTFace*Nvec*precision;
776  size_t spitch = stride*Nvec*precision;//ndeg tm: stride=2*flavor_volume+pad
777  size_t dpitch = 2*len;
778  qudaMemcpy2DAsync(dst, dpitch, src, spitch, len, Npad, cudaMemcpyDeviceToHost, *stream);
779  dst = (char*)ghost_spinor+len;
780  src = (char*)v + flavor2_offset*Nvec*precision;
781  qudaMemcpy2DAsync(dst, dpitch, src, spitch, len, Npad, cudaMemcpyDeviceToHost, *stream);
782 
783  if (precision == QUDA_HALF_PRECISION) {
784  int Nt_minus1_offset = (flavorVolume - flavorTFace);
785  int norm_offset = (dir == QUDA_BACKWARDS) ? 0 : Nt_minus1_offset*sizeof(float);
786  void *dst = (char*)ghost_spinor + Nint*ghostFace[3]*precision;
787  void *src = (char*)norm + norm_offset;
788  size_t dpitch = flavorTFace*sizeof(float);
789  size_t spitch = flavorVolume*sizeof(float);
790  qudaMemcpy2DAsync(dst, dpitch, src, spitch, flavorTFace*sizeof(float), 2, cudaMemcpyDeviceToHost, *stream);
791  }
792  }
793 #else
794  errorQuda("sendGhost not built on single-GPU build");
795 #endif
796
